\subsection*{General Random Variables}

\kwub{Random variable} => \stu{measurable function} \iMbox{X: \Omega \to E}
\begin{enumerate}
    \vItem \iMbox{(\Omega, \mathcal{F}, \mathrm{P})} is a \stu{probability space},
    \iMbox{(E, \mathcal{E})} is a \stu{measurable space}

    \vItem For every \iMbox{B \in \mathcal{E}} the pre-image of \iMbox{B} under \iMbox{X}
    is in \iMbox{\mathcal{F}},

    i.e. \iMbox{X^{-1}[B] = \{ s \in \Omega \mid X(s) \in B \} \in \mathcal{F}}

    i.e. \iMbox{\sigma(X) \subseteq \mathcal{F}} where \iMbox{\sigma(X)} is generated by \ul{function} \iMbox{X}

    \vItem \iMbox{g(X)(s) = (g \circ X)(s)} is \stu{also random variable},
           for \stu{measurable function} \iMbox{g: E \to E} 
\end{enumerate}

\kwub{Induced probability} \iMbox{\mathrm{P}_X(X \in B)} =>
probability that \iMbox{X} takes on value in \iMbox{B \in \mathcal{E}}
\begin{enumerate}
    \vItem \iMbox{\mathrm{P}_X(X \in B) = \mathrm{P}(X^{-1}[B]) = \mathrm{P}(\{ s \in \Omega \mid X(s) \in B \})}
    \vItem Also called \kwb{pushforward measure} of \iMbox{P} onto \iMbox{(E, \mathcal{E})} induced by \iMbox{X} \ => \
           \iMbox{(E, \mathcal{E}, \mathrm{P}_X)} is a \stu{probability space}


    \vItem Also called the \kwub{probability distribution} of \iMbox{X}
\end{enumerate}

\subsection*{Real Random Variables}

\kwub{Real random variable} is \stu{RV} who's \ul{co-domain} is \iMbox{E = \mathbb{R}} 
\begin{enumerate}
    \vItem \kwb{Support} \iMbox{\mathrm{supp}(X)} => is range of \iMbox{X} i.e. \iMbox{\mathrm{supp}(X) = X[\Omega]}
    \begin{enumerate}
        \vItem \kwub{Simple RRV} \ \stub[orange]{iff} \ \stu{finite} \iMbox{\mathrm{supp}(X)}
        \vItem \kwub{Discrete RRV} \ \stub[orange]{iff} \ \stu{countable} \iMbox{\mathrm{supp}(X)}
        \vItem \kwub{Continuous RRV} => \stu{uncountable} \iMbox{\mathrm{supp}(X)}
    \end{enumerate}

    \vItem \kwb{Induced probability} => 
    \iMbox{\mathrm{P}_X(X \leq x) = \mathrm{P}(\{ s \in \Omega \mid X(s) \leq x \})}
\end{enumerate}

\hSep

\kwub{cumulative distribution func. (CDF)} \iMbox{F_X(x) = \mathrm{P}_X(X \leq x)}
\begin{enumerate}
    \vItem \iMbox{F_X(x)} is \stu{right-continuous} => for any \ul{decreasing} \iMbox{\langle x_n \rangle},
    \iMbox{\lim_{n \to \infty} x_n = x_L \implies \lim_{n \to \infty} F_X(x_n) = F_X(x_L)}

    \vItem To \ul{check that function is valid CDF}, must obey:
    \begin{enumerate}
        \vItem \kwu{Monotonicity}
        \iMbox{\forall x_{1}, x_{2} \in \mathbb{R}, x_{1}<x_{2} \implies F_{X}(x_{1}) \leq F_{X}(x_{2})}
        \vItem \iMbox{F_{X}(-\infty)=0, F_{X}(\infty)=1}
        \vItem \iMbox{F_{X}} is right-continuous
    \end{enumerate}

    \vItem \textbf{Simple Properties}
    \begin{enumerate}
        \vItem \iMbox{0 \leq F_{X}(x) \leq 1, \forall x \in \mathbb{R}}
        \vItem \iMbox{\mathrm{P}_{X}(a<X \leq b)=F_{X}(b)-F_{X}(a)} for 
        finite intervals \iMbox{(a, b] \subseteq \mathbb{R}}

    \end{enumerate}
\end{enumerate}
